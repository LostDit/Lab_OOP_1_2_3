class Radio {
private:
    unsigned int volume;         // текущая громкость
    unsigned int station;        // текущая станция

    // Константы границ значений
    static const unsigned int MIN_VOLUME = 0;
    static const unsigned int MAX_VOLUME = 10;
    static const unsigned int MIN_STATION = 1;
    static const unsigned int MAX_STATION = 10;

public:
    // Конструктор по умолчанию
    Radio();

    // Геттеры
    unsigned int getVolume() const;
    unsigned int getStation() const;

    // Сеттеры с проверкой допустимых значений
    void setVolume(unsigned int const & volume);
    void setStation(unsigned int const & station);

    // Деструктор
    ~Radio();
};

// Реализация методов

Radio::Radio() : volume(3), station(MIN_STATION) {}

unsigned int Radio::getVolume() const {
    return volume;
}

unsigned int Radio::getStation() const {
    return station;
}

void Radio::setVolume(unsigned int const & volume) {
    // Если новое значение в допустимых пределах, меняем поле
    if (volume >= MIN_VOLUME && volume <= MAX_VOLUME) {
        this->volume = volume;
    }
}

void Radio::setStation(unsigned int const & station) {
    if (station >= MIN_STATION && station <= MAX_STATION) {
        this->station = station;
    }
}

Radio::~Radio() {
    // Деструктор ничего не делает, так как нет динамических ресурсов
}
